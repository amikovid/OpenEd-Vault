    <script>
    function toggleTranscript() {
    const button = document.querySelector('.transcript-toggle');
    const content = document.getElementById('transcriptContent');
    const buttonText = button.querySelector('.toggle-text');
    
    button.classList.toggle('active');
    content.classList.toggle('show');
    
    if (content.classList.contains('show')) {
    buttonText.textContent = 'Hide Transcript';
    } else {
    buttonText.textContent = 'View Transcript';
    }
    }
    
    // Function to render markdown
    function renderMarkdown() {
    const transcriptDiv = document.getElementById('transcriptText');
    const markdownContent = transcriptDiv.textContent;
    
    // Configure marked options
    marked.setOptions({
    breaks: true, // Converts single line breaks to <br>
    gfm: true, // GitHub Flavored Markdown
    });
    
    // Render the markdown
    transcriptDiv.innerHTML = marked.parse(markdownContent);
    }
    
    // Render markdown when page loads
    document.addEventListener('DOMContentLoaded', renderMarkdown);
    </script>
